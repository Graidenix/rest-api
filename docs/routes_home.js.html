routes/home.js

"use strict";
/**
 * @module routes/home
 * @author godajiu
 */

/**
 * Main app singleton
 * @inner
 * @type {express}
 */
var express = require('express');

/**
 * Router
 * @type {Router}
 * @name router
 */
var router = express.Router();


/**
 * @alias router.root
 */
router.get("/", function (req, res) {

    /**
     *
     * @type {HomeController}
     */
    var HomeController = require("../controllers/home");

    /**
     *
     * @type {HomeController}
     */
    var home = new HomeController(req);


    res.send(home.index()).end();
});



/**
 * Home Route
 */
module.exports = router;
